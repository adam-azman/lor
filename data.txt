CREATE TABLE students (
    registration_number VARCHAR(50) PRIMARY KEY,
    email_id VARCHAR(100) UNIQUE NOT NULL,
    student_name VARCHAR(100) NOT NULL,
    gender VARCHAR(10),
    mobile_number VARCHAR(15),
    date_of_birth DATE NOT NULL
);
